X = []
PX = []
cx = 0
CX = []
x = int(input('For how many coins you want to create probability distribution table? \n'))
print('x p(x)')
for i in range (x+1):
    print(i,scipy.special.comb(x,i)/2**x)
    X.append(i)
    px = scipy.special.comb(x,i)/2**x
    cx = cx + px
    CX.append(cx)
    PX.append(px)
